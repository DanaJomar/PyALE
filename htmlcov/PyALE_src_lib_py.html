7def cmds(D, k=2): 

8 """Classical multidimensional scaling 

9  

10 Theory and code references: 

11 https://en.wikipedia.org/wiki/Multidimensional_scaling#Classical_multidimensional_scaling 

12 http://www.nervouscomputer.com/hfs/cmdscale-in-python/ 

13 

14 Arguments: 

15 D -- A squared matrix-like object (array, DataFrame, ....), usually a distance matrix 

16 """ 

17 

18 n = D.shape[0] 

19 if D.shape[0] != D.shape[1] : 

20 raise Exception("The matrix D should be squared") 

21 if k > (n - 1): 

22 raise Exception("k should be an integer <= D.shape[0] - 1") 

23 

24 # (1) Set up the squared proximity matrix 

25 D_double = np.square(D) 

26 # (2) Apply double centering: using the centering matrix 

27 # centering matrix 

28 center_mat = np.eye(n) - np.ones((n, n)) / n 

29 # apply the centering 

30 B = -(1 / 2) * center_mat.dot(D_double).dot(center_mat) 

31 # (3) Determine the m largest eigenvalues 

32 # (where m is the number of dimensions desired for the output) 

33 # extract the eigenvalues 

34 eigenvals, eigenvecs = np.linalg.eigh(B) 

35 # sort descending 

36 idx = np.argsort(eigenvals)[::-1] 

37 eigenvals = eigenvals[idx] 

38 eigenvecs = eigenvecs[:, idx] 

39 # (4) Now, X=eigenvecs.dot(eigen_sqrt_diag), 

40 # where eigen_sqrt_diag = diag(sqrt(eigenvals)) 

41 eigen_sqrt_diag = np.diag(np.sqrt(eigenvals[0:k])) 

42 ret = eigenvecs[:, 0:k].dot(eigen_sqrt_diag) 

43 return ret

46def order_groups(X, feature): 

47 """Assign an order to the values of a categorical feature.  

48  

49 The function returns an order to the unique values in X[feature] according to  

50 their similarity based on the other features. 

51 The distance between two categories is the sum over the distances of each feature. 

52  

53 Arguments: 

54 X -- A pandas DataFrame containing all the features to considering in the ordering  

55 (including the categorical feature to be ordered). 

56 feature -- String, the name of the column holding the categorical feature to be ordered. 

57 """ 

58 

59 features = X.columns 

60 # groups = X[feature].cat.categories.values 

61 groups = X[feature].unique() 

62 D_cumu = pd.DataFrame(0, index=groups, columns=groups) 

63 K = len(groups) 

64 for j in set(features) - set([feature]): 

65 D = pd.DataFrame(index=groups, columns=groups) 

66 # discrete/factor feature j 

67 # e.g. j = 'color' 

68 if (X[j].dtypes.name == "category") | ( 

69 (len(X[j].unique()) <= 10) & ("float" not in X[j].dtypes.name) 

70 ): 

71 # counts and proportions of each value in j in each group in 'feature' 

72 cross_counts = pd.crosstab(X[feature], X[j]) 

73 cross_props = cross_counts.div(np.sum(cross_counts, axis=1), axis=0) 

74 for i in range(K): 

75 group = groups[i] 

76 D_values = abs(cross_props - cross_props.loc[group]).sum(axis=1) / 2 

77 D.loc[group, :] = D_values 

78 D.loc[:, group] = D_values 

79 else: 

80 # continuous feature j 

81 # e.g. j = 'length' 

82 # extract the 1/100 quantiles of the feature j 

83 seq = np.arange(0, 1, 1 / 100) 

84 q_X_j = X[j].quantile(seq).to_list() 

85 # get the ecdf (empiricial cumulative distribution function) 

86 # compute the function from the data points in each group 

87 X_ecdf = X.groupby(feature)[j].agg(ECDF) 

88 # apply each of the functions on the quantiles 

89 # i.e. for each quantile value get the probability that j will take 

90 # a value less than or equal to this value. 

91 q_ecdf = X_ecdf.apply(lambda x: x(q_X_j)) 

92 for i in range(K): 

93 group = groups[i] 

94 D_values = q_ecdf.apply(lambda x: max(abs(x - q_ecdf[group]))) 

95 D.loc[group, :] = D_values 

96 D.loc[:, group] = D_values 

97 D_cumu = D_cumu + D 

98 # reduce the dimension of the cumulative distance matrix to 1 

99 D1D = cmds(D_cumu, 1).flatten() 

100 # order groups based on the values 

101 order_idx = D1D.argsort() 

102 groups_ordered = D_cumu.index[D1D.argsort()] 

103 return pd.Series(range(K), index=groups_ordered)

106def quantile_ied(x_vec, q): 

107 """ 

108 Inverse of empirical distribution function (quantile R type 1). 

109  

110 More details in 

111 https://docs.scipy.org/doc/scipy/reference/generated/scipy.stats.mstats.mquantiles.html 

112 https://stat.ethz.ch/R-manual/R-devel/library/stats/html/quantile.html  

113 https://en.wikipedia.org/wiki/Quantile 

114  

115 Arguments: 

116 x_vec -- A pandas series containing the values to compute the quantile for 

117 q -- An array of probabilities (values between 0 and 1) 

118 """ 

119 

120 x_vec = x_vec.sort_values() 

121 n = len(x_vec) - 1 

122 m = 0 

123 j = (n * q + m).astype(int) # location of the value 

124 g = n * q + m - j 

125 

126 gamma = (g != 0).astype(int) 

127 quant_res = (1 - gamma) * x_vec.shift(1, fill_value=0).iloc[j] + gamma * x_vec.iloc[ 

128 j 

129 ] 

130 quant_res.index = q 

131 # add min at quantile zero and max at quantile one (if needed) 

132 if 0 in q: 

133 quant_res.loc[0] = x_vec.min() 

134 if 1 in q: 

135 quant_res.loc[1] = x_vec.max() 

136 return quant_res

139def CI_estimate(x_vec, C=0.95): 

140 """Estimate the size of the confidence interval of a data sample. 

141  

142 The confidence interval of the given data sample (x_vec) is  

143 [mean(x_vec) - returned value, mean(x_vec) + returned value]. 

144 """ 

145 alpha = 1 - C 

146 n = len(x_vec) 

147 stand_err = x_vec.std() / np.sqrt(n) 

148 critical_val = 1 - (alpha / 2) 

149 z_star = stand_err * t.ppf(critical_val, n - 1) 

150 return z_star
