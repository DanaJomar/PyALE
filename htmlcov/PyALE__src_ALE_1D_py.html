10def aleplot_1D_continuous(X, model, feature, grid_size=20, include_CI=True, C=0.95): 

11 """Compute the accumulated local effect of a numeric continuous feature. 

12  

13 This function divides the feature in question into grid_size intervals (bins)  

14 and computes the difference in prediction between the first and last value  

15 of each interval and then centers the results. 

16 

17 Arguments: 

18 X -- A pandas DataFrame to pass to the model for prediction. 

19 model -- Any python model with a predict method that accepts X as input. 

20 feature -- String, the name of the column holding the feature being studied. 

21 grid_size -- An integer indicating the number of intervals into which the  

22 feature range is divided. 

23 include_CI -- A boolean, if True the confidence interval  

24 of the effect is returned with the results.  

25 C -- A float the confidence level for which to compute the confidence interval 

26  

27 Return: A pandas DataFrame containing for each bin: the size of the sample in it 

28 and the accumulated centered effect of this bin. 

29 """ 

30 

31 quantiles = np.append(0, np.arange(1 / grid_size, 1 + 1 / grid_size, 1 / grid_size)) 

32 # use customized quantile function to get the same result as 

33 # type 1 R quantile (Inverse of empirical distribution function) 

34 bins = [X[feature].min()] + quantile_ied(X[feature], quantiles).to_list() 

35 bins = np.unique(bins) 

36 feat_cut = pd.cut(X[feature], bins, include_lowest=True) 

37 

38 bin_codes = feat_cut.cat.codes 

39 bin_codes_unique = np.unique(bin_codes) 

40 

41 X1 = X.copy() 

42 X2 = X.copy() 

43 X1[feature] = [bins[i] for i in bin_codes] 

44 X2[feature] = [bins[i + 1] for i in bin_codes] 

45 y_1 = model.predict(X1).ravel() 

46 y_2 = model.predict(X2).ravel() 

47 

48 delta_df = pd.DataFrame({feature: bins[bin_codes + 1], "Delta": y_2 - y_1}) 

49 res_df = delta_df.groupby([feature]).Delta.agg([("eff", "mean"), "size"]) 

50 res_df["eff"] = res_df["eff"].cumsum() 

51 res_df.loc[min(bins), :] = 0 

52 # subtract the total average of a moving average of size 2 

53 mean_mv_avg = ( 

54 (res_df["eff"] + res_df["eff"].shift(1, fill_value=0)) / 2 * res_df["size"] 

55 ).sum() / res_df["size"].sum() 

56 res_df = res_df.sort_index().assign(eff=res_df["eff"] - mean_mv_avg) 

57 if include_CI: 

58 ci_est = delta_df.groupby(feature).Delta.agg( 

59 [("CI_estimate", lambda x: CI_estimate(x, C=C))] 

60 ) 

61 ci_est = ci_est.sort_index() 

62 lowerCI_name = "lowerCI_" + str(int(C * 100)) + "%" 

63 upperCI_name = "upperCI_" + str(int(C * 100)) + "%" 

64 res_df[lowerCI_name] = res_df[["eff"]].subtract(ci_est["CI_estimate"], axis=0) 

65 res_df[upperCI_name] = upperCI = res_df[["eff"]].add( 

66 ci_est["CI_estimate"], axis=0 

67 ) 

68 return res_df

71def aleplot_1D_discrete(X, model, feature, include_CI=True, C=0.95): 

72 """Compute the accumulated local effect of a numeric discrete feature. 

73  

74 This function computes the difference in prediction when the value of the feature 

75 is replaced once with the value before it and once with the value after it, without  

76 the need to divide into interval like the case of aleplot_1D_continuous. 

77 

78 Arguments: 

79 X -- A pandas DataFrame to pass to the model for prediction. 

80 model -- Any python model with a predict method that accepts X as input. 

81 feature -- String, the name of the column holding the feature being studied. 

82 include_CI -- A boolean, if True the confidence interval  

83 of the effect is returned with the results.  

84 C -- A float the confidence level for which to compute the confidence interval 

85  

86 Return: A pandas DataFrame containing for each value of the feature: the size  

87 of the sample in it and the accumulated centered effect around this value. 

88 """ 

89 

90 groups = X[feature].unique() 

91 groups.sort() 

92 groups_codes = [x for x in range(len(groups))] 

93 

94 groups_counts = X.groupby(feature).size() 

95 groups_props = groups_counts / sum(groups_counts) 

96 

97 K = len(groups) 

98 

99 # create copies of the dataframe 

100 X_plus = X.copy() 

101 X_neg = X.copy() 

102 # all groups except last one 

103 ind_plus = X[feature] < groups[K - 1] 

104 # all groups except first one 

105 ind_neg = X[feature] > groups[0] 

106 # replace once with one level up 

107 X_plus.loc[ind_plus, feature] = groups[X.loc[ind_plus, feature] + 1] 

108 # replace once with one level down 

109 X_neg.loc[ind_neg, feature] = groups[X.loc[ind_neg, feature] - 1] 

110 # predict with original and with the replaced values 

111 y_hat = model.predict(X) 

112 y_hat_plus = model.predict(X_plus[ind_plus]) 

113 y_hat_neg = model.predict(X_neg[ind_neg]) 

114 # compute prediction difference 

115 Delta_plus = y_hat_plus - y_hat[ind_plus] 

116 Delta_neg = y_hat[ind_neg] - y_hat_neg 

117 

118 # compute the mean of the difference per group 

119 delta_df = pd.concat( 

120 [ 

121 pd.DataFrame({"eff": Delta_plus, feature: X.loc[ind_plus, feature] + 1}), 

122 pd.DataFrame({"eff": Delta_neg, feature: X.loc[ind_neg, feature]}), 

123 ] 

124 ) 

125 res_df = delta_df.groupby([feature]).mean() 

126 res_df["eff"] = res_df["eff"].cumsum() 

127 res_df.loc[0] = 0 

128 res_df = res_df.sort_index() 

129 res_df["eff"] = res_df["eff"] - sum(res_df["eff"] * groups_props) 

130 res_df["size"] = groups_counts 

131 if include_CI: 

132 ci_est = delta_df.groupby([feature]).eff.agg( 

133 [("CI_estimate", lambda x: CI_estimate(x, C=C))] 

134 ) 

135 lowerCI_name = "lowerCI_" + str(int(C * 100)) + "%" 

136 upperCI_name = "upperCI_" + str(int(C * 100)) + "%" 

137 res_df[lowerCI_name] = res_df[["eff"]].subtract(ci_est["CI_estimate"], axis=0) 

138 res_df[upperCI_name] = upperCI = res_df[["eff"]].add( 

139 ci_est["CI_estimate"], axis=0 

140 ) 

141 return res_df

144def plot_1D_continuous_eff(res_df, X, fig=None, ax=None): 

145 """Plot the 1D ALE plot for a continuous feature. 

146  

147 Arguments: 

148 res_df -- A pandas DataFrame containing the computed effects  

149 (the output of ale_1D_continuous). 

150 X -- The dataset used to compute the effects. 

151 fig, ax -- matplotlib figure and axis. 

152 """ 

153 

154 feature_name = res_df.index.name 

155 # position: jitter 

156 # to see the distribution of the data points clearer, each point x will be nudged a random value between 

157 # -0.5*(distance from the bin's lower value) and +0.5*(distance from bin's upper value) 

158 jitter_limits = pd.DataFrame( 

159 { 

160 "x": X[feature_name], 

161 "bin_code": pd.cut( 

162 X[feature_name], res_df.index.to_list(), include_lowest=True 

163 ).cat.codes 

164 + 1, 

165 } 

166 ).assign( 

167 jitter_step_min=lambda df: (df["x"] - res_df.index[df["bin_code"] - 1]) * 0.5, 

168 jitter_step_max=lambda df: (res_df.index[df["bin_code"]] - df["x"]) * 0.5, 

169 ) 

170 np.random.seed(123) 

171 rug = jitter_limits.apply( 

172 lambda row: row["x"] 

173 + np.random.uniform(-row["jitter_step_min"], row["jitter_step_max"]), 

174 axis=1, 

175 ) 

176 

177 if fig is None and ax is None: 

178 fig, ax = plt.subplots(figsize=(8, 4)) 

179 ax.plot(res_df[["eff"]]) 

180 tr = mtrans.offset_copy(ax.transData, fig=fig, x=0.0, y=-5, units="points") 

181 ax.plot( 

182 rug, 

183 [res_df.drop("size", axis=1).min().min()] * len(rug), 

184 "|", 

185 color="k", 

186 alpha=0.2, 

187 transform=tr, 

188 ) 

189 lowerCI_name = res_df.columns[res_df.columns.str.contains("lowerCI")] 

190 upperCI_name = res_df.columns[res_df.columns.str.contains("upperCI")] 

191 if (len(lowerCI_name) == 1) and (len(upperCI_name) == 1): 

192 label = lowerCI_name.str.split("_")[0][1] + " confidence interval" 

193 ax.fill_between( 

194 res_df.index, 

195 y1=res_df[lowerCI_name[0]], 

196 y2=res_df[upperCI_name[0]], 

197 alpha=0.2, 

198 color="grey", 

199 label=label, 

200 ) 

201 ax.legend() 

202 ax.set_xlabel(res_df.index.name) 

203 ax.set_ylabel("Effect on prediction (centered)") 

204 ax.set_title("1D ALE Plot - Continuous") 

205 return fig, ax

208def plot_1D_discrete_eff(res_df, X, fig=None, ax=None): 

209 """Plot the 1D ALE plot for a discrete feature. 

210  

211 Arguments: 

212 res_df -- A pandas DataFrame with the computed effects

214 X -- The dataset used to compute the effects.
